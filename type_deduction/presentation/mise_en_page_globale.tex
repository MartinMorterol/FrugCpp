% la gestion des sommaires
\AtBeginSection[]
{
 \begin{frame}
  \frametitle{Sommaire}
  \tableofcontents[currentsection, hideothersubsections]
  \end{frame} 
}


\AtBeginSubsection[]
{
 \begin{frame}
  \frametitle{Sommaire}
  \tableofcontents[  
    sectionstyle=show/shaded,
    subsectionstyle=show/shaded/hide
    ]
  \end{frame} 
}
